618  USB->PCGCCTL |= USB_PCGCCTL_STOPPCLK;

626 #endif

632 #else

634  while ( ( CMU->STATUS & CMU_STATUS_USBCLFRCOSEL ) == 0 ){}
635 #endif
636 
637  return true;
638  }
